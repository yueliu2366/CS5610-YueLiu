Experiment Description:

In this experiment, I try out $http.get to get JSon format data from server.js. I also use AngularJS to parse data.

Here is the sinppet of JS file:


var app = angular.module("MoviesApp", []);
app.controller("MovieController",
function ($scope, $http) {
  
    $http.get("/api/movies").success(function (response) {
        $scope.movies = response;
    })
   
});

Here is the sinppet of server.js file:


app.get("/api/movies", function (req, res) {
    res.json(movies);
});
